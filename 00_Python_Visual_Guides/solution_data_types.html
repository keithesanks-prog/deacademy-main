import pandas as pd

# 1. Setup Data
data = {
    'Product': ['Laptop', 'Phone', 'Tablet', 'Monitor'],
    'Price': [1000.5, 500.75, 300.25, 150.99],
    'Quantity': [5, 10, 15, 7],
    'Date': ['2023-01-10', '2022-11-15', '2021-06-21', '2020-03-18']
}

df = pd.DataFrame(data)

print('--- Before Conversion ---')
print(df)
print("\nTypes:")
print(df.dtypes)

# 2. Convert Price to Integer
# We use astype(int) to force truncation because to_numeric won't allow data loss (decimals)
df['Price'] = df['Price'].astype(int)

# 3. Convert Date to Datetime
df['Date'] = pd.to_datetime(df['Date'])

# 4. Convert Quantity to String
df['Quantity'] = df['Quantity'].astype('str')

print('\n--- After Conversion ---')
print(df)
print("\nNew Types:")
print(df.dtypes)
